[Desktop Entry]
Encoding=UTF-8
Type=Application
Exec=stumpwm --replace
TryExec=stumpwm
Name=StumpWM
Comment=Stump Window Manager

## Remap stumpwm key to Control space, turn on modeline
~/.stumpwmrc
;; Remap prefix
(in-package :stumpwm)
(set-prefix-key (kbd "C-SPC"))

;; Switch mode-line on
(if (not (head-mode-line (current-head)))
    (toggle-mode-line (current-screen) (current-head)))

## Remap caps lock to control
~/.Xmodmap
!
! Remap Caps to Control
!
remove Lock = Caps_Lock
remove Control = Control_L
keysym Caps_Lock = Control_L
add Control = Control_L
